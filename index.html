// Logic for fetching latest music releases function fetchLatestReleases() { const delay = Math.random() * (1500 - 1000) + 1000; // random delay between 1s and 1.5s setTimeout(async () => { try { const albumsResponse = await fetch('API_ENDPOINT_FOR_ALBUMS'); const singlesResponse = await fetch('API_ENDPOINT_FOR_SINGLES'); const albums = await albumsResponse.json(); const singles = await singlesResponse.json(); // Filter albums to 2 latest const latestAlbums = albums.filter(album => album.type === 'album').slice(0, 2); // Filter singles to 5 latest const latestSingles = singles.filter(single => single.type === 'single').slice(0, 5); // Process and display latest albums and singles displayReleases(latestAlbums, latestSingles); } catch (error) { console.error('Error fetching releases:', error); } }, delay); } fetchLatestReleases();
